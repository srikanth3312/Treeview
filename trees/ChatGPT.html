To achieve this in WPF using C#, you can create a custom control or user control that contains two TreeView controls. Then, you can handle events such as button clicks to send data between them. Here's a basic example to get you started:

  1. Define your XAML layout for the custom control (DualTreeViewControl.xaml):
xml
<UserControl x:Class="YourNamespace.DualTreeViewControl"
             xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
             xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml">
    <Grid>
        <Grid.ColumnDefinitions>
            <ColumnDefinition Width="*" />
            <ColumnDefinition Width="Auto" />
            <ColumnDefinition Width="*" />
        </Grid.ColumnDefinitions>

        <TreeView x:Name="treeViewLeft" Grid.Column="0" />
        <StackPanel Grid.Column="1" Orientation="Vertical">
            <Button Content="&lt;" Click="OnSendLeftToRightClicked" />
            <Button Content="&gt;" Click="OnSendRightToLeftClicked" />
        </StackPanel>
        <TreeView x:Name="treeViewRight" Grid.Column="2" />
    </Grid>
</UserControl>
  1. Implement the code-behind logic (DualTreeViewControl.xaml.cs):
csharp
using System.Windows;
using System.Windows.Controls;

namespace YourNamespace
{
    public partial class DualTreeViewControl : UserControl
    {
        public DualTreeViewControl()
        {
            InitializeComponent();
        }

        public TreeView LeftTreeView => treeViewLeft;
        public TreeView RightTreeView => treeViewRight;

        private void OnSendLeftToRightClicked(object sender, RoutedEventArgs e)
        {
            // Move selected item from left to right
            if (treeViewLeft.SelectedItem != null)
            {
                var selectedItem = treeViewLeft.SelectedItem;
                treeViewLeft.Items.Remove(selectedItem);
                treeViewRight.Items.Add(selectedItem);
            }
        }

        private void OnSendRightToLeftClicked(object sender, RoutedEventArgs e)
        {
            // Move selected item from right to left
            if (treeViewRight.SelectedItem != null)
            {
                var selectedItem = treeViewRight.SelectedItem;
                treeViewRight.Items.Remove(selectedItem);
                treeViewLeft.Items.Add(selectedItem);
            }
        }
    }
}
